\begin{algorithm}
\begin{algorithmic}[1]
\caption{Insertion Sort}
\Procedure {Insert-Sort}{$A$}\Comment{$Worst-Case: \theta{(n^2)}$}
\For {$j \gets 2 \ \textbf{to} \ A.length$}
  \State {$key \gets A[j]$}
  \State {$i \gets j - 1$}
  \While{$i\geq0 \land A[i]>key$}
    \State A[i+1] = A[i]
    \State {$i \gets i - 1$}
  \EndWhile
  \State A[i+1]=key
\EndFor
\State {$\textbf{return}$\ {A}}
\EndProcedure
\end{algorithmic}
\end{algorithm}

\begin{algorithm}
\begin{algorithmic}[1]
\caption{Selection Sort}
\Procedure{Select-Sort}{$A$}\Comment{$Worst-Case: \theta{(n^2)}$}
\For {$i \gets 1\ \textbf{to}\ A.length-1$}
  \State {$min \gets i$}
  \For {$j \gets i+1\ \textbf{to}\ A.length$}
    \If {$A[j] < A[min]$}
      \State $min \gets j$
      \EndIf
  \EndFor
  \State $key \gets A[i]$
  \State $A[i] \gets A[min]$
  \State $A[min] \gets key$
\EndFor
\State {$\textbf{return}$\ {A}}
\EndProcedure
\end{algorithmic}
\end{algorithm}
